86 inline void sort()
87 {
88 if(mSize < 1)return;
89 if(dynamic_cast<Comparable<T>*>(&mData[0]) == nullptr)return;
90
91 size_t n = mSize;
92 do
93 {
94 size_t newn = 1;
95 for(size_t i = 0; i < n - 1; ++i)
96 {
97 size_t j = i + 1;
98 Comparable<T>* a1 = static_cast<Comparable<T>*>(&mData[i]);
99 Comparable<T>* a2 = static_cast<Comparable<T>*>(&mData[j]);
100 if(a1->compareTo(*static_cast<T*>(a2)) > 0)
101 {
102 //Vertausche
103 T tmp = mData[i];
104 mData[i] = mData[j];
105 mData[j] = tmp;
106
107 newn = i + 1;
108 }
109 }
110 n = newn;
111 }
112 while(n > 1);
113 }
